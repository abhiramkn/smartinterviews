C Pattern Programming Notes

1. Square Pattern

Output

*****
*****
*****
*****
*****
Logic
  • Rows = n
  • Columns = n
  • Print '*' every time

Formula

Rows = n
Stars = n

Table

Row Stars
15
25
35
45
55

C Code

#include <stdio.h>

int main()
{
    int n;
    scanf("%d",&n);

    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
        {
            printf("*");
        }
        printf("\n");
    }

    return 0;
}

4. Right Triangle

Output

    *
   **
  ***
 ****
*****
Logic
  • Spaces = n-i
  • Stars = i
Row Spaces Stars
141
232
323
414
505

C Code

#include <stdio.h>

int main()
{
    int n;
    scanf("%d",&n);

    for(int i=1;i<=n;i++)
    {

        for(int j=1;j<=n-i;j++)
        {
            printf(" ");
        }

        for(int j=1;j<=i;j++)
        {
            printf("*");
        }

        printf("\n");

    }

    return 0;
}
